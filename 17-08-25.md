#COUNT ODD NUMBERS
numbers = [10, 21, 32, 43, 54, 65, 76, 87]
print("Odd numbers in the list are:")
for num in numbers:
    if num % 2 != 0:   
        print(num)

#PRIME NUMBERS IN RANGE
start = 10
end = 50
print("Prime numbers between", start, "and", end, ":")
for num in range(start, end + 1):
    if num > 1:
        for i in range(2, num):
            if num % i == 0:
                break
        else:
            print(num, end=" ")
